当数字为偶数时 1的数量等会 i/2的中1的数量  i <<<1
当数字为奇数时 1的数量等会 (i-1)为偶数 +1
//f(n) = n%2==0? f(n/2):f(n-1)+1
```java
class Solution {
    public int[] countBits(int num) {
        int[] dp = new int[num + 1];
        dp[0] = 0;
        for (int i = 1; i <= num; i++) {
            if (i % 2 == 0) {
                dp[i] = dp[i / 2];
            } else {
                dp[i] = dp[i - 1] + 1;
            }
        }
        return dp;
    }
}
```
