0,1背包问题
通过总数和/2 特判断
f(i,j) = f(i-1,j) || f(i-1,j-k(i))
```java
class Solution {
    public boolean canPartition(int[] nums) {
        int count = 0;
        for (int num : nums) {
            count += num;
        }
        if (count % 2 == 1) {
            return false;
        }
        count = count / 2;
        boolean[][] dp = new boolean[nums.length][count + 1];
        dp[0][0] = true;
        if (nums[0] == count) {
            dp[0][nums[0]] = true;
        }

        for (int i = 1; i < nums.length; i++) {
            for (int j = 0; j <= count; j++) {
                dp[i][j] = dp[i - 1][j];
                if (nums[i] <= j) {
                    dp[i][j] = dp[i - 1][j] || dp[i - 1][j - nums[i]];
                }
            }
        }
        return dp[nums.length - 1][count];
    }
}
```
