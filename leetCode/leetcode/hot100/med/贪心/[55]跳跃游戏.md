class Solution {
    public boolean canJump(int[] nums) {
        int rightMax = 0;
        for (int i = 0; i < nums.length; i++) {
            if (i <= rightMax) {
                rightMax = Math.max(rightMax, nums[i] + i);
                if(rightMax >= nums.length -1){
                    return true;
                }
            }
        }
        return false;
    }
}
