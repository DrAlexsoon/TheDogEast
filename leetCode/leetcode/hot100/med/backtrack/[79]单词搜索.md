class Solution {
    public boolean exist(char[][] board, String word) {
        for (int i = 0; i < board.length; i++) {
            for (int j = 0; j < board[0].length; j++) {
                if (board[i][j] == word.charAt(0)) {
                    if (bfs(board, word, i, j, 0)) {
                        return true;
                    }
                }
            }
        }
        return false;
    }

    boolean bfs(char[][] board, String word, int i, int j, int charAt) {
        if(charAt == word.length()){
            return true;
        }
        if(i >=0 && j>=0 && i<board.length && j < board[0].length){
            if(board[i][j] == word.charAt(charAt)){
                char tmp = board[i][j];
                board[i][j]= '?';
                boolean b = bfs(board, word, i - 1, j, charAt + 1) ||
                        bfs(board, word, i + 1, j, charAt + 1) ||
                        bfs(board, word, i, j - 1, charAt + 1) ||
                        bfs(board, word, i, j + 1, charAt + 1);
                board[i][j] = tmp;
                if(b){
                    return b;
                }
            }
        }
        return false;
    }

}
