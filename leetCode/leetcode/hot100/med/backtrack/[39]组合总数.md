class Solution {
    LinkedList<List<Integer>> result;

    public List<List<Integer>> combinationSum(int[] candidates, int target) {
        result = new LinkedList<List<Integer>>();
        LinkedList<Integer> linkedList = new LinkedList<>();
        combinationSum(candidates, target, 0, linkedList);
        return result;
    }

    void combinationSum(int[] candidates, int target, int i, LinkedList<Integer> linkedList) {

        if (target < 0) {
            return;
        }
        if (target == 0) {
            result.add(new ArrayList<>(linkedList));
        } else {
            for (int j = i; j < candidates.length; j++) {
                linkedList.add(candidates[j]);
                combinationSum(candidates, target - candidates[j], j, linkedList);
                linkedList.pollLast();
            }
        }
    }
}
