- if(left == n&& right == n )
- if(left < n ) '('++ ; 递归;reset;
- if(right < left) ')'++ ;递归;reset;
```java
class Solution {
    List<String> result = new LinkedList<>();
    StringBuilder sb = new StringBuilder();

    public List<String> generateParenthesis(int n) {
        generateParenthesis(0,0,n);
        return result;
    }

    void generateParenthesis(int left, int right, int n) {
        if (left == n && right == n) {
            result.add(sb.toString());
        }
        if (left < n) {
            sb.append('(');
            generateParenthesis(left + 1, right, n);
            sb.setLength(sb.length() - 1);
        }
        if (right < left) {
            sb.append(')');
            generateParenthesis(left, right +1, n);
            sb.setLength(sb.length() - 1);
        }
    }
}
```
