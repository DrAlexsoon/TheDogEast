**2. 左右扩展范围圈定**
```java
class Solution {
    public int[] searchRange(int[] nums, int target) {
        int i = 0;
        int j = nums.length - 1;
        while (i <= j) {
            int midIndex = (i + j) / 2;

            if (target == nums[midIndex]) {
                int m = midIndex;
                int n = midIndex;
                while (m >= 0 && nums[m] == target ) {
                    m--;
                }
                while (n < nums.length &&nums[n] == target ) {
                    n++;
                }
                return new int[]{m + 1, n - 1};
            }
            if (target < nums[midIndex]) {
                j = midIndex - 1;
            } else {
                i = midIndex + 1;
            }
        }
        return new int[]{-1, -1};
    }
}
```
