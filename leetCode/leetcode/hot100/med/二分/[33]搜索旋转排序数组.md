1.查找中间值mid
2.if(mid >= start ) 升序|
    if(target 在[start,mid] ){
        end = mid -1;
    }else{
        start = mid+1;
    }
3.if(mid <= start) |升序
    if(target 在[mid,end] 区间){
        start = mid+1;
    }else{
        end = mid -1;
    }
```java
class Solution {
    //[4,5,6,7,0,1,2] 0
    //[4,5,6,7,0,1,2 9
    public int search(int[] nums, int target) {
        //
        int i = 0;
        int j = nums.length - 1;

        while (i <= j) {
            int midIndex = (i + j) / 2;
            if (nums[midIndex] == target) {
                return midIndex;
            }
            if (nums[midIndex] >= nums[i]) {
                if (target >= nums[i] && target < nums[midIndex]) {
                    j = midIndex - 1;
                } else {
                    i = midIndex + 1;
                }
            } else {
                if (target <= nums[j] && target > nums[midIndex]) {
                    i = midIndex + 1;
                } else {
                    j = midIndex - 1;
                }
            }

        }
        return -1;
    }

}
```
