#数组复制 + index 
```java
class Solution {
    public boolean isPalindrome(ListNode head) {
        ArrayList<ListNode> arrayList = new ArrayList<>();
        while (head != null) {
            arrayList.add(head);
            head = head.next;
        }
        int i = 0;
        int j = arrayList.size() - 1;
        while(i < j){
            if(arrayList.get(i).val == arrayList.get(j).val){
                i++;
                j--;
            }else{
                return false;
            }
        }
        return true;
    }
}
```

#递归
```java
class Solution {
    ListNode fornNode;

    public boolean isPalindrome(ListNode head) {
        fornNode = head;
        return check(head);
    }
    boolean check(ListNode head) {
        if (head != null) {
            if (!check(head.next)) {
                return false;
            }
            if (head.val != fornNode.val) {
                return false;
            }
            fornNode = fornNode.next;
        }
        return true;
    }
}
```
