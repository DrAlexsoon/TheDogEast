class MinStack {
    Stack<Integer> stack ;
    Stack<Integer> min;
    /** initialize your data structure here. */
    public MinStack() {
        stack =  new Stack<>();
        min = new Stack<>();
    }

    public void push(int x) {
        if(stack.isEmpty()){
            min.push(x);
        }else{
            min.push(Math.min(x,min.peek()));
        }
        stack.push(x);
    }

    public void pop() {
        min.pop();
        stack.pop();
    }

    public int top() {
        return stack.peek();
    }

    public int getMin() {
        return min.peek();
    }
}
