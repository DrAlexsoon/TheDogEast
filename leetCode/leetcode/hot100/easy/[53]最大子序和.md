//f(n) = Math.max(f(n),f(n-1))
class Solution {
    public int maxSubArray(int[] nums) {
        //f(n,m) =
        int current = Integer.MIN_VALUE;
        int result = Integer.MIN_VALUE;
        for (int i = 0; i < nums.length; i++) {
            if (current < 0 && nums[i] > current) {
                current = nums[i];
            } else {
                current += nums[i];
            }
            result = Math.max(current,result);
        }
        return result;
    }
}
