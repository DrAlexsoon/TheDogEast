class Solution {
    public List<Integer> findDisappearedNumbers(int[] nums) {
        LinkedList<Integer> linkedList = new LinkedList<>();
        for (int i = 0; i < nums.length; i++) {
            int newIndex = Math.abs(nums[i]) - 1;
            if (nums[newIndex] > 0) {
                nums[newIndex] = nums[newIndex] * (-1);
            }
        }
        for (int i = 0; i < nums.length; i++) {
            if(nums[i] >0){
                linkedList.add(i+1);
            }
        }
        return linkedList;
    }
}
