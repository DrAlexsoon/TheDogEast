public class Solution {
    public ListNode getIntersectionNode(ListNode headA, ListNode headB) {
        int la = 0;
        int lb = 0;
        ListNode couserA = headA;
        ListNode couserB = headB;
        while(couserA !=null){
            la++;
            couserA = couserA.next;
        }
        while(couserB !=null){
            lb++;
            couserB = couserB.next;
        }
        if(la < lb){
            int step = lb -la;
            for (int i = 0; i < step; i++) {
                headB = headB.next;
            }
        }else if(la > lb){
            int step = la -lb;
            for (int i = 0; i < step; i++) {
                headA = headA.next;
            }
        }
        while(headA !=null){
            if(headA == headB){
                return headA;
            }
            headA = headA.next;
            headB = headB.next;
        }
        return null;
    }
}
