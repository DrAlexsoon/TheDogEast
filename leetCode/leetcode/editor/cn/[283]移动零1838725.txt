class Solution {
    public void moveZeroes(int[] nums) {
        int count = 0;
        for(int i =0; i<nums.length;i++){
            if(nums[i]==0){
                count++;
            }else{
                int tmp = nums[i];
                nums[i-count] = tmp;
            }
        }
        for(int j = count;j>0;j--){
            nums[nums.length-j] =0;
        }
    }
}
