class Solution {
    public int cuttingRope(int n) {
        if(n <= 3){
            return n-1;
        }
        long result = 1;
        while(n > 4){
           n -=3;
           result = result*3 %1000000007;
        }
        return (int)(result*n%1000000007);
    }
}
