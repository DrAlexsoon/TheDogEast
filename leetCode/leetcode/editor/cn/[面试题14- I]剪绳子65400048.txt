class Solution {
    public int cuttingRope(int n) {
        if(n <= 3){
            return n -1;
        }
        int pow = 0;
        while(n >= 3){
            n -=3;
            pow++; 
        }
        if(n == 0){
            return (int)Math.pow(3,pow);
        }else if(n == 1){
            return (int)Math.pow(3,pow-1) * 4;
        }else {
            return (int)Math.pow(3,pow) * 2;
        }
    }
}
