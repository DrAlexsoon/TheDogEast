class Solution {
    public boolean wordBreak(String s, List<String> wordDict) {
        if(s.length() == 0){
            return true;
        }
        HashSet<String> words = new HashSet<>(wordDict);
        for (int i = 0; i < wordDict.size(); i++) {
            String pre = wordDict.get(i);
            if (s.startsWith(pre)) {
                if (wordBreak(s.substring(pre.length()), wordDict)) {
                    return true;
                }
            }
        }
        return false;
    }
}
