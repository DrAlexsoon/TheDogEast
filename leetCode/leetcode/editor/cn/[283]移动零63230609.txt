class Solution {
    public void moveZeroes(int[] nums) {
        //Arrays.sort(nums);
        int count = 0;
        for(int i = 0; i< nums.length ;i++){
            if(nums[i] == 0){
                count++;
            }else{
                nums[i-count] = nums[i];
            }
        }
        int end = nums.length ;
        while(count !=0){
            nums[--end] = 0;
            count--;
        }
    }
}
